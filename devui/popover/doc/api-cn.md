# 如何使用

在 module 中引入：

```ts
import { PopoverModule } from 'ng-devui/popover';
```

在页面中使用：

```html
<xxx dPopover></xxx>
```

# 接口 & 类型定义

### PositionType

```ts
export type PositionType =
  | 'left'
  | 'right'
  | 'top'
  | 'bottom'
  | 'bottom-left'
  | 'bottom-right'
  | 'top-left'
  | 'top-right'
  | 'left-top'
  | 'left-bottom'
  | 'right-top'
  | 'right-bottom';
```
