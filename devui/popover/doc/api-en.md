# How to use

Import into module:

```ts
import { PopoverModule } from 'ng-devui/popover';
```

In the page:

```html
<xxx dPopover></xxx>
```

# Interface & Type Definition

### PositionType

```ts
export type PositionType =
  | 'left'
  | 'right'
  | 'top'
  | 'bottom'
  | 'bottom-left'
  | 'bottom-right'
  | 'top-left'
  | 'top-right'
  | 'left-top'
  | 'left-bottom'
  | 'right-top'
  | 'right-bottom';
```
